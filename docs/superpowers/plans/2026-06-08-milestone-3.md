# Milestone 3 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Polish the game to shipping quality — particles, pause, vignette, mobile touch controls, and bomb cap.

**Architecture:** Each feature is a self-contained addition: particle emitters hang off existing objects/collision events; pause is a new overlaid Phaser scene; touch controls live in a standalone system class; vignette and bomb cap are single-line fixes.

**Tech Stack:** Phaser 3.87 (new particles API v3.60+, scene.pause/resume, postFX), vanilla JS ES modules, Vite.

**Already confirmed done — do not re-implement:**
- High score save + display (localStorage in GameScene.showDeathOverlay + MainMenuScene)
- Boss death cinematic (10 staggered explosions in Boss.defeat())
- Boss Phase 2 minion spawning (Boss._spawnMinions())
- Per-level scroll speed variation (ScrollManager L1=0.5, L2=0.7, L3=0.3–1.3)
- Difficulty scaling (DIFFICULTY_PRESETS + gradual ramp)

import { GAME_WIDTH, GAME_HEIGHT } from '../constants.js';

export default class PauseScene extends Phaser.Scene {
  constructor() { super('PauseScene'); }

  create() {
    const cx = GAME_WIDTH / 2, cy = GAME_HEIGHT / 2;

    // Semi-transparent overlay
    this.add.rectangle(cx, cy, GAME_WIDTH, GAME_HEIGHT, 0x000000, 0.55).setDepth(0);

    this.add.text(cx, cy - 60, 'PAUSED', {
      font: 'bold 48px monospace', fill: '#ffffff',
      stroke: '#004488', strokeThickness: 5
    }).setOrigin(0.5).setDepth(1);

    const resume = this.add.text(cx, cy + 20, '[ P  —  RESUME ]', {
      font: 'bold 20px monospace', fill: '#44aaff',
      stroke: '#001133', strokeThickness: 3
    }).setOrigin(0.5).setDepth(1).setInteractive({ useHandCursor: true });

    const menu = this.add.text(cx, cy + 65, '[ M  —  MAIN MENU ]', {
      font: '16px monospace', fill: '#aaaaaa',
      stroke: '#000', strokeThickness: 2
    }).setOrigin(0.5).setDepth(1).setInteractive({ useHandCursor: true });

    this.tweens.add({ targets: resume, alpha: 0.4, duration: 550, yoyo: true, repeat: -1 });

    const doResume = () => {
      this.scene.resume('GameScene');
      this.scene.resume('HUDScene');
      this.scene.stop();
    };

    const doMenu = () => {
      this.scene.stop('GameScene');
      this.scene.stop('HUDScene');
      this.scene.stop();
      this.scene.start('MainMenuScene');
    };

    resume.on('pointerdown', doResume);
    menu.on('pointerdown',   doMenu);
    this.input.keyboard.once('keydown-P', doResume);
    this.input.keyboard.once('keydown-ESC', doResume);
    this.input.keyboard.once('keydown-M', doMenu);
  }
}

### 5a: TouchControls system

- [ ] **Step 1: Create `src/systems/TouchControls.js`**

```javascript
import { GAME_WIDTH } from '../constants.js';

// Split-screen touch: left half = joystick movement, right half = fire.
// Player.update() reads this._touchVelX / this._touchVelY / this._touchFire.
export default class TouchControls {
  constructor(scene, player) {
    this.scene  = scene;
    this.player = player;

    this._leftPointer  = null;  // pointer id for movement side
    this._rightPointer = null;  // pointer id for fire side
    this._originX = 0;
    this._originY = 0;
    this._velX = 0;
    this._velY = 0;
    this._firing = false;

    // Expose on player for Player.update() to read
    player._touchVelX  = 0;
    player._touchVelY  = 0;
    player._touchFire  = false;

    scene.input.on('pointerdown',  this._onDown,  this);
    scene.input.on('pointermove',  this._onMove,  this);
    scene.input.on('pointerup',    this._onUp,    this);
    scene.input.on('pointercancel',this._onUp,    this);
  }

  _onDown(pointer) {
    if (pointer.x < GAME_WIDTH / 2) {
      if (this._leftPointer !== null) return;
      this._leftPointer = pointer.id;
      this._originX = pointer.x;
      this._originY = pointer.y;
      this._velX = 0;
      this._velY = 0;
    } else {
      if (this._rightPointer !== null) return;
      this._rightPointer = pointer.id;
      this._firing = true;
    }
    this._sync();
  }

  _onMove(pointer) {
    if (pointer.id !== this._leftPointer) return;
    const dx = pointer.x - this._originX;
    const dy = pointer.y - this._originY;
    const deadzone = 12;
    const speed = this.player.speed;
    const len = Math.sqrt(dx * dx + dy * dy);
    if (len < deadzone) {
      this._velX = 0;
      this._velY = 0;
    } else {
      const clamped = Math.min(len, 60);
      this._velX = (dx / len) * speed * (clamped / 60);
      this._velY = (dy / len) * speed * (clamped / 60);
    }
    this._sync();
  }

  _onUp(pointer) {
    if (pointer.id === this._leftPointer) {
      this._leftPointer = null;
      this._velX = 0;
      this._velY = 0;
    }
    if (pointer.id === this._rightPointer) {
      this._rightPointer = null;
      this._firing = false;
    }
    this._sync();
  }

  _sync() {
    this.player._touchVelX = this._velX;
    this.player._touchVelY = this._velY;
    this.player._touchFire = this._firing;
  }

  destroy() {
    this.scene.input.off('pointerdown',  this._onDown, this);
    this.scene.input.off('pointermove',  this._onMove, this);
    this.scene.input.off('pointerup',    this._onUp,   this);
    this.scene.input.off('pointercancel',this._onUp,   this);
  }
}
```
